------------------------------------------------------------- # دالة جلب وقراءة ملف القنوات العربية من GitHub # ------------------------------------------------------------- def fetch_arabic_channels(): url = "https://iptv-org.github.io/iptv/languages/ara.m3u" channels = {} print("🌐 جاري سحب القنوات العربية من GitHub...") try: req = urllib.request.Request(url, headers={'User-Agent': 'Mozilla/5.0'}) with urllib.request.urlopen(req) as response: content = response.read().decode('utf-8') lines = content.split('\n') current_name = None for line in lines: line = line.strip() if line.startswith('#EXTINF:'): match = re.search(r',([^,]+)$', line) if match: current_name = match.group(1).strip() elif line.startswith('http') and current_name: channels[current_name] = line current_name = None print(f"✅ تم تحميل {len(channels)} قناة عربية بنجاح!") return channels except Exception as e: print(f"❌ فشل جلب القنوات بسبب: {e}") return {"قناة تجريبية تعمل دائماً (MP4)": "https://storage.googleapis.com/gtv-videos-bucket/sample/BigBuckBunny.mp4"} # ------------------------------------------------------------- # كود تطبيق التلفزيون الرئيسي المعتمد على VLC # ------------------------------------------------------------- class SmartTVApp(QWidget): def __init__(self): super().__init__() self.setWindowTitle("تطبيق التلفزيون العربي بمحرك VLC 📺") self.setGeometry(100, 100, 1200, 700) self.setStyleSheet("background-color: #0c0f12;") self.channels = fetch_arabic_channels() self.init_ui() def init_ui(self): main_layout = QHBoxLayout() main_layout.setContentsMargins(0, 0, 0, 0) main_layout.setSpacing(0) # تصميم قائمة القنوات الجانبية self.channel_list = QListWidget() self.channel_list.addItems(self.channels.keys()) self.channel_list.setFixedWidth(350) self.channel_list.setStyleSheet(""" QListWidget { background-color: rgba(20, 24, 33, 245); color: #e2e8f0; font-size: 16px; font-family: 'Segoe UI', 'Arial'; font-weight: bold; border: none; padding: 10px; } QListWidget::item { padding: 12px; margin-bottom: 4px; border-radius: 6px; color: #cbd5e1; } QListWidget::item:hover { background-color: #1e293b; color: #ffffff; } QListWidget::item:selected { background-color: #2563eb; color: #ffffff; } """) self.channel_list.itemClicked.connect(self.play_channel) # إنشاء حاوية فارغة للفيديو (تمهيداً لربطها بـ VLC) self.video_container = QWidget() self.video_container.setStyleSheet("background-color: #000000;") main_layout.addWidget(self.channel_list) main_layout.addWidget(self.video_container) self.setLayout(main_layout) # تهيئة مشغل VLC داخل واجهة بايثون self.vlc_instance = vlc.Instance() self.media_player = self.vlc_instance.media_player_new() # ربط مخرج فيديو VLC بنوافذ بايثون (تحديداً لويندوز) if sys.platform.startswith('win'): self.media_player.set_hwnd(int(self.video_container.winId())) self.channel_list.setFocus() if self.channel_list.count() > 0: self.channel_list.setCurrentRow(0) def play_channel(self, item): channel_name = item.text() url = self.channels[channel_name] print(f"🎬 تشغيل عبر VLC: {channel_name} -> {url}") # إنشاء ميديا وتشغيلها في مشغل VLC media = self.vlc_instance.media_new(url) self.media_player.set_media(media) self.media_player.play() def keyPressEvent(self, event): if event.key() == Qt.Key_Right or event.key() == Qt.Key_Space: if self.channel_list.isVisible(): self.channel_list.hide() else: self.channel_list.show(); self.channel_list.setFocus() elif event.key() == Qt.Key_Escape or event.key() == Qt.Key_Left: self.channel_list.hide() else: super().keyPressEvent(event) def closeEvent(self, event): # إغلاق مشغل VLC بأمان عند قفل البرنامج self.media_player.stop() super().closeEvent(event) if __name__ == "__main__": app = QApplication(sys.argv) tv_app = SmartTVApp() tv_app.show() sys.exit(app.exec_())
