import subprocess import sys import urllib.request import re # ------------------------------------------------------------- # التحقق من وجود مكتبة PyQt5 وتثبيتها تلقائياً إن نقصت # ------------------------------------------------------------- try: from PyQt5 import QtWidgets from PyQt5.QtMultimedia import QMediaPlayer, QMediaContent from PyQt5.QtMultimediaWidgets import QVideoWidget from PyQt5.QtCore import QUrl, Qt from PyQt5.QtWidgets import QApplication, QWidget, QHBoxLayout, QListWidget except ImportError: print("🔔 جاري تثبيت مكتبة PyQt5 لتشغيل التلفزيون... يرجى الانتظار ثواني...") try: subprocess.check_call([sys.executable, "-m", "pip", "install", "PyQt5"]) print("✅ تم التثبيت بنجاح! جاري تشغيل التطبيق...") from PyQt5.QtMultimedia import QMediaPlayer, QMediaContent from PyQt5.QtMultimediaWidgets import QVideoWidget from PyQt5.QtCore import QUrl, Qt from PyQt5.QtWidgets import QApplication, QWidget, QHBoxLayout, QListWidget except Exception as e: print(f"❌ فشل التثبيت التلقائي بسبب: {e}") sys.exit(1) # ------------------------------------------------------------- # دالة ذكية لتحميل وقراءة ملف M3U من الإنترنت (M3U Parser) # ------------------------------------------------------------- def fetch_arabic_channels(): url = "https://iptv-org.github.io/iptv/languages/ara.m3u" channels = {} print("🌐 جاري تحميل قائمة القنوات العربية من الإنترنت...") try: # الاتصال بالرابط وقراءة المحتوى req = urllib.request.Request(url, headers={'User-Agent': 'Mozilla/5.0'}) with urllib.request.urlopen(req) as response: content = response.read().decode('utf-8') # تقسيم الملف إلى أسطر لقراءتها lines = content.split('\n') current_name = None for line in lines: line = line.strip() if line.startswith('#EXTINF:'): # استخراج اسم القناة الذي يأتي بعد الفاصلة الأخيرة في السطر match = re.search(r',([^,]+)$', line) if match: current_name = match.group(1).strip() elif line.startswith('http') and current_name: # إذا كان السطر يحتوي على رابط، نربطه بالاسم المستخرج channels[current_name] = line current_name = None # إعادة التعيين للقناة القادمة print(f"✅ تم تحميل {len(channels)} قناة عربية بنجاح!") return channels except Exception as e: print(f"❌ فشل جلب القنوات بسبب: {e}") # قنوات احتياطية في حال انقطع الإنترنت أو حُجب الرابط return {"⚠️ فشل التحميل - قناة تجريبية": "https://demo.unified-streaming.com/k8s/features/stable/video/tears-of-steel/tears-of-steel.ism/.m3u8"} # ------------------------------------------------------------- # كود تطبيق التلفزيون الرئيسي # ------------------------------------------------------------- class SmartTVApp(QWidget): def __init__(self): super().__init__() self.setWindowTitle("تطبيق القنوات العربية المباشرة 📺") self.setGeometry(100, 100, 1200, 700) self.setStyleSheet("background-color: #0c0f12;") # جلب القنوات تلقائياً عند تشغيل التطبيق self.channels = fetch_arabic_channels() self.init_ui() def init_ui(self): main_layout = QHBoxLayout() main_layout.setContentsMargins(0, 0, 0, 0) main_layout.setSpacing(0) # تصميم قائمة القنوات الجانبية المحدثة self.channel_list = QListWidget() self.channel_list.addItems(self.channels.keys()) self.channel_list.setFixedWidth(350) # عريضة قليلاً لتناسب الأسماء العربية self.channel_list.setStyleSheet(""" QListWidget { background-color: rgba(20, 24, 33, 245); color: #e2e8f0; font-size: 16px; font-family: 'Segoe UI', 'Arial'; font-weight: bold; border: none; padding: 10px; } QListWidget::item { padding: 12px; margin-bottom: 4px; border-radius: 6px; color: #cbd5e1; text-align: right; /* محاذاة النص للغة العربية */ } QListWidget::item:hover { background-color: #1e293b; color: #ffffff; } QListWidget::item:selected { background-color: #2563eb; color: #ffffff; } """) self.channel_list.itemClicked.connect(self.play_channel) # إنشاء منطقة مشغل الفيديو self.video_widget = QVideoWidget() self.video_widget.setStyleSheet("background-color: #000000;") self.media_player = QMediaPlayer(None, QMediaPlayer.VideoSurface) self.media_player.setVideoOutput(self.video_widget) main_layout.addWidget(self.channel_list) main_layout.addWidget(self.video_widget) self.setLayout(main_layout) self.channel_list.setFocus() if self.channel_list.count() > 0: self.channel_list.setCurrentRow(0) def play_channel(self, item): channel_name = item.text() url = self.channels[channel_name] print(f"🎬 جاري تشغيل: {channel_name} -> {url}") self.media_player.setMedia(QMediaContent(QUrl(url))) self.media_player.play() def keyPressEvent(self, event): # التحكم بالريموت أو الكيبورد (إخفاء وإظهار القائمة) if event.key() == Qt.Key_Right or event.key() == Qt.Key_Space: if self.channel_list.isVisible(): self.channel_list.hide() else: self.channel_list.show() self.channel_list.setFocus() elif event.key() == Qt.Key_Escape or event.key() == Qt.Key_Left: self.channel_list.hide() else: super().keyPressEvent(event) if __name__ == "__main__": app = QApplication(sys.argv) tv_app = SmartTVApp() # إذا أردت تشغيل التطبيق بملء الشاشة فوراً كالتلفزيون احذف علامة # من السطر التالي: # tv_app.showFullScreen() tv_app.show() sys.exit(app.exec_())
